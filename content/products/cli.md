				<div class="col-xs-12 col-md-6">
					<div class="landingPageOutlineHeading">

					</div>

                    </p>
				</div>
				<div class="col-xs-12 col-md-6">
					<div class="landingPageOutlineHeading">
						Configure embedded devices
					</div>

                    </p>
				</div>
				<div class="col-xs-12 col-md-6">
					<div class="landingPageOutlineHeading">
						Monitor embedded hardware from the cloud
					</div>
